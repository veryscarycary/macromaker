The test file lives at `__tests__/screens/DietTodayScreen.test.tsx` — two levels from the project root. All mock paths use `../../` (not `../../../../`). Confirmed by the existing `BasicInfoScreen.test.tsx` in the same directory which uses `../../context/InfoContext`.

```typescript
import React from 'react';
import { render } from '@testing-library/react-native';

// Mock BarGraph to capture the data prop
jest.mock('../../components/BarGraph', () => {
  const React = require('react');
  const BarGraphMock = (props: any) => {
    BarGraphMock.lastProps = props;
    return null;
  };
  BarGraphMock.lastProps = null;
  return BarGraphMock;
});

// Mock TotalCaloriesGraph and MealTimeGraph (not under test)
jest.mock('../../components/TotalCaloriesGraph', () => () => null);
jest.mock('../../components/MealTimeGraph', () => () => null);
jest.mock('../../components/MealTimeGraph/utils', () => ({
  getMealTimeMealsWithColor: jest.fn(() => []),
}));

// Mock async storage deps
jest.mock('../../utils', () => ({
  ...jest.requireActual('../../utils'),
  getStoredData: jest.fn(() => Promise.resolve(null)),
  getTodaysDate: jest.fn(() => '01/01/2025'),
}));
jest.mock('../../context/MealContext', () => ({
  getMealData: jest.fn(() => Promise.resolve(null)),
}));

import { colors } from '../../design/tokens/colors';
import DietTodayScreen from '../../screens/Diet/screens/Today/DietTodayScreen';
const BarGraphMock = require('../../components/BarGraph') as any;

describe('DietTodayScreen', () => {
  it('passes colors.macro.* tokens to BarGraph data', () => {
    render(<DietTodayScreen navigation={{ navigate: jest.fn() } as any} />);
    const data = BarGraphMock.lastProps?.data ?? [];
    const carbsItem = data.find((d: any) => d.label === 'Carbs');
    const proteinItem = data.find((d: any) => d.label === 'Protein');
    const fatItem = data.find((d: any) => d.label === 'Fat');
    expect(carbsItem?.color).toBe(colors.macro.carbs);
    expect(proteinItem?.color).toBe(colors.macro.protein);
    expect(fatItem?.color).toBe(colors.macro.fat);
  });
});
```
